## GPIO setup for pin 29 in, pin 31 32 33 out
/dts-v1/;
/plugin/;

/ {
    jetson-header-name = "Jetson 40pin Header";
    overlay-name = "29in313233out";
    compatible = "nvidia,p3768-0000+p3767-0005-super",
                 "nvidia,p3767-0005",
                 "nvidia,tegra234";

    fragment@0 {
        target = <&pinmux>;

        __overlay__ {
            pinctrl-names = "default";
            pinctrl-0 = <&jetson_io_pinmux>;

            jetson_io_pinmux: exp-header-pinmux {
                hdr40-pin31 {
                    nvidia,pins = "soc_gpio33_pq6";
                    nvidia,tristate = <0x0>;        // output (driven)
                    nvidia,enable-input = <0x1>;
                    nvidia,pull = <0x0>;            
                };

                hdr40-pin29 {
                    nvidia,pins = "soc_gpio32_pq5";
                    nvidia,tristate = <0x1>;        // input (Hi-Z)
                    nvidia,enable-input = <0x1>;
                    nvidia,pull = <0x2>;
                };

                hdr40-pin32 {
                    nvidia,pins = "soc_gpio12_pb4";
                    nvidia,tristate = <0x0>;        // output (driven)
                    nvidia,enable-input = <0x0>;
                    nvidia,pull = <0x0>;            // no pull for output
                };

                hdr40-pin33 {
                    nvidia,pins = "soc_gpio13_pb5";
                    nvidia,tristate = <0x0>;        // output (driven)
                    nvidia,enable-input = <0x0>;
                    nvidia,pull = <0x0>;            // no pull for output
                };
            };
        };
    };
};

## DOCKER setup
Make persistent container
docker run -it \
  --name sam2 \
  --restart unless-stopped \
  --runtime nvidia \
  --network host \
  --shm-size=8g \
  transformers:5.2.0-r36.4.tegra-aarch64-cu126-22.04-onnx \
  bash

Bring container back
docker start -ai sam2

Start container
docker exec -it sam2 bash

stop/delete
docker stop/rm <image name>
