## Run OAK Camera (https://docs.luxonis.com/hardware/platform/deploy/to-jetson/)

sudo reboot now
# Disable ZRAM:
sudo systemctl disable nvzramconfig
# Create 4GB swap file
sudo fallocate -l 4G /mnt/4GB.swap
sudo chmod 600 /mnt/4GB.swap
sudo mkswap /mnt/4GB.swap
sudo vi /etc/fstab
# Add this line at the bottom of the file
/mnt/4GB.swap swap swap defaults 0 0
# Reboot
sudo reboot now
sudo wget -qO- https://docs.luxonis.com/install_dependencies.sh | bash

## DOCKER setup
Make persistent container
docker run -it \
  --name sam2 \
  --restart unless-stopped \
  --runtime nvidia \
  --network host \
  --shm-size=8g \
  transformers:5.2.0-r36.4.tegra-aarch64-cu126-22.04-onnx \
  bash

Bring container back
docker start -ai sam2

Start container
docker exec -it sam2 bash

stop/delete
docker stop/rm <image name>

Run jupyter notebook
Jupyter notebook –allow-root (in the directory)
